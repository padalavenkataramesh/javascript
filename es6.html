//ES6 for beginners: //-------------- /** ** ** Let and Const Arrow functions Default parameters for of loop Spread attributes Maps Sets Static methods Getters and Setters ** ** */ ----------------------------------------------- 1) Let: block level scop variable. i) if (true) { let a = 40; console.log(a); //40 } console.log(a); // undefined ii) let a = 50; let b = 100; if (true) { let a = 60; var c = 10; console.log(a/c); // 6 console.log(b/c); // 10 } console.log(c); // 10 console.log(a); // 50 2) Const: -> Const is used to assign a constant value to the variable. And the value cannot be changed. Its fixed i) const a = 50; a = 60; // shows error. You cannot change the value of const. const b = "Constant variable"; b = "Assigning new value"; // shows error. ii) const LANGUAGES = ['Js', 'Ruby', 'Python', 'Go']; LANGUAGES = "Javascript"; // shows error. LANGUAGES.push('Java'); // Works fine. console.log(LANGUAGES); // ['Js', 'Ruby', 'Python', 'Go', 'Java']; 3) Arrow Function:- in es6 to describe methods little bit difference differnt approch. i) // Old Syntax function oldOne() { console.log("Hello World..!"); } // New Syntax var newOne = () => { console.log("Hello World..!"); } ii) let NewOneWithParameters = (a, b) => { console.log(a+b); // 30 } NewOneWithParameters(10, 20); 4) Default Parameters:Default parameters are parameters which are given by default while declaring a function. But it’s value can be changed when calling the function. i) let Func = (a, b = 10) => { return a + b; } Func(20); // 20 + 10 = 30 Func(20, 50); // 20 + 50 = 70 ii) let NotWorkingFunction = (a = 10, b) => { return a + b; } NotWorkingFunction(20); // NAN. Not gonna work. 5) For of loop:for..of is very similar to for..in with slight modification, for..of iterates through list of elements (i.e) like Array and returns the elements (not their index) one by one. i) let arr = [2,3,4,1]; for (let value of arr) { console.log(value); } Output: 2 3 4 1 ii) let string = "Javascript"; for (let char of string) { console.log(char); } Output: J a v a s c r i p t 6) Spread attributes: Spread attributes help to spread the expression as the name suggests. In simple words, it converts a list of elements to an array and vice versa. i)without spread attributes: let SumElements = (arr) => { console.log(arr); // [10, 20, 40, 60, 90] let sum = 0; for (let element of arr) { sum += element; } console.log(sum); // 220. } SumElements([10, 20, 40, 60, 90]); ii) let SumElements = (...arr) => { console.log(arr); // [10, 20, 40, 60, 90] let sum = 0; for (let element of arr) { sum += element; } console.log(sum); // 220. } SumElements(10, 20, 40, 60, 90); // Note we are not passing array here. Instead we are passing the elements as arguments. iii) vise versa - converts the array to list of elements. Math.max(10, 20, 60, 100, 50, 200); // returns 200. let arr = [10, 20, 60]; Math.max(arr); // Shows error. Doesn't accept an array. let arr = [10, 20, 60]; Math.max(...arr); // 60 7) Maps:-Map holds key-value pairs. It’s similar to an array and also we can define our own index. And indexes are unique in maps. i) var NewMap = new Map(); NewMap.set('name', 'John'); NewMap.set('id', 2345796); NewMap.set('interest', ['js', 'ruby', 'python']); NewMap.get('name'); // John NewMap.get('id'); // 2345796 NewMap.get('interest'); // ['js', 'ruby', 'python'] ii) Other interesting features of Maps are all indexes are unique. And we can use any value as key or value. var map = new Map(); map.set('name', 'John'); map.set('name', 'Andy'); map.set(1, 'number one'); map.set(NaN, 'No value'); map.get('name'); // Andy. Note John is replaced by Andy. map.get(1); // number one map.get(NaN); // No value iii) var map = new Map(); map.set('name', 'John'); map.set('id', 10); map.size; // 2. Returns the size of the map. map.keys(); // outputs only the keys. map.values(); // outputs only the values. for (let key of map.keys()) { console.log(key); } Output: name id iv) var map = new Map(); map.set('name', 'John'); map.set('id', 10); for (let [key, value] of map) { console.log(key+" - "+value); } 8) Sets:- Sets are used to store the unique values of any type i) var sets = new Set(); sets.add('a'); sets.add('b'); sets.add('a'); // We are adding duplicate value. for (let element of sets) { console.log(element); } Output: a b ii) var sets = New Set([1,5,6,8,9]); sets.size; // returns 5. Size of the size. ( size is self-explanatory) sets.has(1); // returns true. sets.has(10); // returns false. 9) Static methods:Static methods are introduced in ES6. And it is pretty much easy to define it and use it. i) Note that I didn’t use the keyword ‘function’ inside Class.And I can call the function without creating any instance for the class. class Example { static Callme() { console.log("Static method"); } } Example.Callme(); Output: Static method 10) Getters and Setters: i) without getters and setters class People { constructor(name) { this.name = name; } getName() { return this.name; } setName(name) { this.name = name; } } let person = new People("Jon Snow"); console.log(person.getName()); person.setName("Dany"); console.log(person.getName()); Output: Jon Snow Dany ii) with getters and setters class People { constructor(name) { this.name = name; } get Name() { return this.name; } set Name(name) { this.name = name; } } let person = new People("Jon Snow"); console.log(person.Name); person.Name = "Dany"; console.log(person.Name);
